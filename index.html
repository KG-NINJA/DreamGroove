export default { async fetch(request, env) { try { // 🔥 あなたのPages URL const allowedOrigin = "https://kg-ninja.github.io"; // CORS preflight if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders(allowedOrigin), }); } if (!env.DREAM_BUCKET) { return new Response("NO BUCKET", { status: 500, headers: corsHeaders(allowedOrigin), }); } if (request.method !== "POST") { return new Response("NG", { status: 405, headers: corsHeaders(allowedOrigin), }); } // 🔥 Referer制限（ここ重要） const referer = request.headers.get("referer") || ""; if (!referer.startsWith("https://kg-ninja.github.io/DreamGroove")) { return new Response("Forbidden: bad referer", { status: 403, headers: corsHeaders(allowedOrigin), }); } // APIキー const apiKey = request.headers.get("x-api-key"); if (apiKey !== env.API_KEY) { return new Response("Unauthorized", { status: 401, headers: corsHeaders(allowedOrigin), }); } // JSON取得 let body; try { body = await request.json(); } catch { return new Response("JSON ERROR", { status: 400, headers: corsHeaders(allowedOrigin), }); } const content = body.content || ""; const tag = body.tag || "voice"; const timestamp = body.timestamp || new Date().toISOString(); if (!content.trim()) { return new Response("EMPTY CONTENT", { status: 400, headers: corsHeaders(allowedOrigin), }); } // ファイル名 const safeTimestamp = timestamp.replace(/[:.]/g, "-"); const filename = `${tag}/${safeTimestamp}.md`; const fullContent = `# Voice Note ${content} --- time: ${timestamp} tag: ${tag} source: iphone `; await env.DREAM_BUCKET.put(filename, fullContent); return new Response( JSON.stringify({ status: "OK", file: filename, }), { status: 200, headers: { "Content-Type": "application/json", ...corsHeaders(allowedOrigin), }, } ); } catch (e) { return new Response("ERROR: " + e.message, { status: 500, headers: corsHeaders("https://kg-ninja.github.io"), }); } }, }; function corsHeaders(origin) { return { "Access-Control-Allow-Origin": origin, "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, x-api-key", }; }
